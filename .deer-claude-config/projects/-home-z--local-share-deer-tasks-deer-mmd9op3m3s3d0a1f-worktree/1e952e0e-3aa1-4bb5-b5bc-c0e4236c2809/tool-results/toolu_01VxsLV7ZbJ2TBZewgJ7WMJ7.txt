   124→/** Suspend the ink alternate screen, run fn, then restore. */
   125→async function withSuspendedTerminal(
   126→  setSuspended: (v: boolean) => void,
   127→  fn: () => Promise<void>,
   128→): Promise<void> {
   129→  setSuspended(true);
   130→  process.stdout.write("\x1b[?1049l");
   131→  if (process.stdin.setRawMode) process.stdin.setRawMode(false);
   132→  try {
   133→    await fn();
   134→  } finally {
   135→    if (process.stdin.setRawMode) process.stdin.setRawMode(true);
   136→    process.stdout.write("\x1b[?1049h\x1b[2J\x1b[H");
   137→    setSuspended(false);
   138→  }
   139→}

   141→/** Spawn a bash script, check exit code, parse JSON from last stdout line. */
   142→async function runScriptJson<T>(args: string[], cwd: string): Promise<T> {
   143→  const proc = Bun.spawn(args, { cwd, stdout: "pipe", stderr: "pipe" });
   144→  const code = await proc.exited;
   145→  if (code !== 0) {
   146→    const stderr = await new Response(proc.stderr).text();
   147→    throw new Error(`Script failed (exit ${code}): ${stderr.trim().split("\n").pop()}`);
   148→  }
   149→  const stdout = await new Response(proc.stdout).text();
   150→  const lines = stdout.trim().split("\n").filter(Boolean);
   151→  const jsonLine = lines[lines.length - 1];
   152→  if (!jsonLine) throw new Error("Script produced no output");
   153→  return JSON.parse(jsonLine) as T;
   154→}

   235→/** Timeout for the needs-input classification invocation */
   236→const NEEDS_INPUT_TIMEOUT_MS = 15_000;

   238→/**
   239→ * Use a fast LLM call to determine if the agent's last transcript message
   240→ * is asking the user for input/clarification/decisions — works across
   241→ * all languages.
   242→ *
   243→ * Returns true if the agent appears to be blocked on human intervention.
   244→ */
   245→export async function needsHumanInput(transcript: string[]): Promise<boolean> {
   246→  if (transcript.length === 0) return false;
   247→  const last = transcript[transcript.length - 1].trim();
   248→  if (!last) return false;
   249→
   250→  const prompt = [
   251→    "You are a classifier. The following message is the final output from an AI coding agent.",
   252→    "Is a response from the user expected after this message?",
   253→    "For example: the agent asked a question, presented options to choose from,",
   254→    "requested clarification, asked the user to do something, or is otherwise",
   255→    "waiting for input before it can continue.",
   256→    "",
   257→    "Respond with exactly YES or NO. Nothing else.",
   258→    "",
   259→    "<agent-message>",
   260→    last.length > 2000 ? last.slice(-2000) : last,
   261→    "</agent-message>",
   262→  ].join("\n");
   263→
   264→  try {
   265→    const { CLAUDECODE: _, ANTHROPIC_API_KEY: __, ...env } = process.env;
   266→    const proc = Bun.spawn([
   267→      "claude", "-p", "--output-format", "text", "--model", "haiku",
   268→    ], {
   269→      stdin: new Response(prompt),
   270→      stdout: "pipe",
   271→      stderr: "pipe",
   272→      env,
   273→    });
   274→
   275→    const timeout = new Promise<never>((_, reject) =>
   276→      setTimeout(() => {
   277→        proc.kill();
   278→        reject(new Error("needs-input classification timed out"));
   279→      }, NEEDS_INPUT_TIMEOUT_MS),
   280→    );
   281→
   282→    const result = await Promise.race([
   283→      proc.exited.then(async (code) => {
   284→        if (code !== 0) return false;
   285→        const text = (await new Response(proc.stdout).text()).trim().toUpperCase();
   286→        return text === "YES";
   287→      }),
   288→      timeout,
   289→    ]);
   290→
   291→    return result;
   292→  } catch {
   293→    // On any failure, assume no — don't block the normal flow
   294→    return false;
   295→  }
   296→}

   368→    errors.push("No OAuth token — set CLAUDE_CODE_OAUTH_TOKEN or create ~/.claude/agent-oauth-token");

   376→async function setupAgent(
   377→  cwd: string,
   378→  agent: AgentState,
   379→  setAgents: (updater: (prev: AgentState[]) => AgentState[]) => void,
   380→  baseBranch?: string,
   381→): Promise<SandboxMeta> {
   382→  const args = ["bash", join(SCRIPTS_DIR, "setup-sandbox.sh"), cwd, MODEL];
   383→  if (baseBranch) args.push(baseBranch);
   384→
   385→  const proc = Bun.spawn(args, { cwd, stdout: "pipe", stderr: "pipe" });
   386→
   387→  // Stream stderr lines into agent logs in real-time
   388→  const stderrStream = proc.stderr as ReadableStream;
   389→  const reader = stderrStream.getReader();
   390→  const decoder = new TextDecoder();
   391→  let stderrBuf = "";
   392→
   393→  (async () => {
   394→    try {
   395→      while (true) {
   396→        const { done, value } = await reader.read();
   397→        if (done) break;
   398→        stderrBuf += decoder.decode(value, { stream: true });
   399→        const lines = stderrBuf.split("\n");
   400→        stderrBuf = lines.pop()!;
   401→        for (const raw of lines) {
   402→          const line = stripAnsi(raw).trim();
   403→          if (line) {
   404→            appendLog(agent, `[setup] ${line}`);
   405→            agent.lastActivity = truncate(line, 120);
   406→            setAgents((prev) => [...prev]);
   407→          }
   408→        }
   409→      }
   410→      // Flush remaining
   411→      const last = stripAnsi(stderrBuf).trim();
   412→      if (last) {
   413→        appendLog(agent, `[setup] ${last}`);
   414→        agent.lastActivity = truncate(last, 120);
   415→        setAgents((prev) => [...prev]);
   416→      }
   417→    } catch { /* stream closed */ }
   418→  })();
   419→
   420→  const code = await proc.exited;
   421→  if (code !== 0) {
   422→    throw new Error(`Setup failed (exit ${code})`);
   423→  }
   424→  const stdout = await new Response(proc.stdout).text();
   425→  const jsonLines = stdout.trim().split("\n").filter(Boolean);
   426→  const jsonLine = jsonLines[jsonLines.length - 1];
   427→  if (!jsonLine) throw new Error("Setup script produced no output");
   428→  return JSON.parse(jsonLine) as SandboxMeta;
   429→}

   431→/**
   432→ * Apply a deny-by-default network policy to the sandbox, allowing only the
   433→ * domains in the config allowlist. Called after sandbox creation but before
   434→ * the agent starts, so tmux/apt installs during setup are unaffected.

   436→/** Domains that must bypass the MITM proxy so OAuth credentials pass through
   437→ *  unmodified. The `claude` sandbox template's proxy intercepts these and
   438→ *  injects the host's ANTHROPIC_API_KEY, overriding CLAUDE_CODE_OAUTH_TOKEN. */
   439→const PROXY_BYPASS_HOSTS = [
   440→  "api.anthropic.com",
   441→  "claude.ai",
   442→  "statsig.anthropic.com",
   443→  "sentry.io",

   446→async function applyNetworkPolicy(sandboxName: string, allowlist: string[]): Promise<void> {
   447→  const args = [
   448→    "docker", "sandbox", "network", "proxy", sandboxName,
   449→    "--policy", "deny",
   450→    ...allowlist.flatMap((host) => ["--allow-host", host]),
   451→    ...PROXY_BYPASS_HOSTS.flatMap((host) => ["--bypass-host", host]),
   452→  ];
   453→  const proc = Bun.spawn(args, { stdout: "pipe", stderr: "pipe" });
   454→  const code = await proc.exited;
   455→  if (code !== 0) {
   456→    const stderr = await new Response(proc.stderr).text();
   457→    throw new Error(`Network policy failed (exit ${code}): ${stderr.trim()}`);
   458→  }

   461→/**
   462→ * Start Claude inside a tmux session in the sandbox.
   463→ * Writes a launcher script to deerTmpDir (bind-mounted into the sandbox)
   464→ * that bakes in the OAuth token and unsets ANTHROPIC_API_KEY. The script
   465→ * runs as tmux's initial command, so when Claude exits the pane dies
   466→ * (remain-on-exit preserves content for scrollback capture).
   467→ *
   468→ * A script file avoids fragile quoting through multiple shell layers
   469→ * (TypeScript → docker exec → sh → tmux send-keys → tmux shell).
   470→ */
   471→async function startClaudeInTmux(meta: SandboxMeta, prompt: string): Promise<void> {
   472→  const promptPath = join(meta.deerTmpDir, ".agent-prompt");
   473→  await Bun.write(promptPath, prompt);
   474→
   475→  // Write launcher script — runs as tmux session's initial command.
   476→  // The token is written directly into the file, avoiding send-keys quoting.
   477→  // deerTmpDir is inside GIT_DIR which is bind-mounted into the sandbox.
   478→  const launcherPath = join(meta.deerTmpDir, ".agent-launcher.sh");
   479→  await Bun.write(launcherPath, [
   480→    `#!/bin/sh`,
   481→    `unset ANTHROPIC_API_KEY`,
   482→    `export CLAUDE_CODE_OAUTH_TOKEN='${process.env.CLAUDE_CODE_OAUTH_TOKEN}'`,
   483→    `cd "${meta.worktreePath}"`,
   484→    `cat "${meta.deerTmpDir}/.agent-prompt" | claude -p --verbose --dangerously-skip-permissions --model ${MODEL}`,
   485→  ].join("\n") + "\n");
   486→
   487→  const script = [
   488→    `export PATH="$PATH:/usr/bin:/usr/local/bin:/bin"`,
   489→    `export TERM=xterm-256color`,
   490→    `chmod +x ${launcherPath}`,
   491→    `tmux new-session -d -s deer "sh ${launcherPath}"`,
   492→    `tmux set -t deer remain-on-exit on`,
   493→  ].join(" && ");
   494→
   495→  const proc = Bun.spawn([
   496→    "docker", "sandbox", "exec", "--privileged", meta.sandboxName,
   497→    "sh", "-c", script,
   498→  ], {
   499→    stdout: "pipe",
   500→    stderr: "pipe",
   501→  });
   502→
   503→  const code = await proc.exited;
   504→  if (code !== 0) {
   505→    const stderr = await new Response(proc.stderr).text();
   506→    throw new Error(`Failed to start Claude in tmux: ${stderr.trim()}`);
   507→  }
   508→}

   510→/**
   511→ * Build the follow-up prompt that asks the agent to write metadata files.
   512→ * Includes the repo's PR template if one exists.
   513→ */
   514→async function buildMetadataPrompt(worktreePath: string, deerTmpDir: string): Promise<string> {
   515→  let prTemplate = "";
   516→
   517→  const templatePaths = [
   518→    ".github/PULL_REQUEST_TEMPLATE.md",
   519→    ".github/pull_request_template.md",
   520→    "PULL_REQUEST_TEMPLATE.md",
   521→    "pull_request_template.md",
   522→  ];
   523→
   524→  for (const rel of templatePaths) {
   525→    const f = Bun.file(join(worktreePath, rel));
   526→    if (await f.exists()) {
   527→      prTemplate = await f.text();
   528→      break;
   529→    }
   530→  }
   531→
   532→  // Collect the diff so the metadata prompt is self-contained (no --continue needed)
   533→  const diffProc = Bun.spawn(["git", "-C", worktreePath, "diff", "HEAD"], {
   534→    stdout: "pipe",
   535→    stderr: "ignore",
   536→  });
   537→  let diff = await new Response(diffProc.stdout).text();
   538→  await diffProc.exited;
   539→
   540→  // If nothing is unstaged, get the diff against the parent commit instead
   541→  if (!diff.trim()) {
   542→    const logProc = Bun.spawn(["git", "-C", worktreePath, "diff", "HEAD~1..HEAD"], {
   543→      stdout: "pipe",
   544→      stderr: "ignore",
   545→    });
   546→    diff = await new Response(logProc.stdout).text();
   547→    await logProc.exited;
   548→  }
   549→
   550→  // Truncate very large diffs to avoid blowing up the context window
   551→  const MAX_DIFF_CHARS = 80_000;
   552→  if (diff.length > MAX_DIFF_CHARS) {
   553→    diff = diff.slice(0, MAX_DIFF_CHARS) + "\n\n... (diff truncated)";
   554→  }
   555→
   556→  const prSection = prTemplate
   557→    ? `A pull request description following this template:\n\n<pr-template>\n${prTemplate}\n</pr-template>`
   558→    : `A pull request description with a summary of the changes.`;
   559→
   560→  return [
   561→    "Below is the git diff of changes made by an automated agent. Based on this diff, write these three files:",
   562→    "",
   563→    `1. \`${deerTmpDir}/.agent-branch-name\``,
   564→    "   A short kebab-case name for a git branch describing the changes.",
   565→    "   No prefix. Examples: fix-login-validation, add-user-avatar-upload",
   566→    "",
   567→    `2. \`${deerTmpDir}/.agent-commit-message\``,
   568→    "   A conventional git commit message. First line is the subject (<72 chars),",
   569→    "   then a blank line, then an optional body.",
   570→    "",
   571→    `3. \`${deerTmpDir}/.agent-pr-body\``,
   572→    `   ${prSection}`,
   573→    "",
   574→    "Write these three files now. Do nothing else.",
   575→    "",
   576→    "<diff>",
   577→    diff,
   578→    "</diff>",
   579→  ].join("\n");
   580→}

   582→/**
   583→ * Run a follow-up `claude -p` to extract metadata (branch name,
   584→ * commit message, PR body) after the main task finishes. The prompt
   585→ * includes the git diff so it's self-contained.
   586→ *
   587→ * Non-fatal — if this fails or times out, teardown has fallbacks.
   588→ */
   589→async function startClaudeMetadata(meta: SandboxMeta, agent: AgentState): Promise<void> {
   590→  const prompt = await buildMetadataPrompt(meta.worktreePath, meta.deerTmpDir);
   591→  const promptPath = join(meta.deerTmpDir, ".agent-metadata-prompt");
   592→  await Bun.write(promptPath, prompt);
   593→
   594→  const proc = spawnClaudeInSandbox(meta, `${meta.deerTmpDir}/.agent-metadata-prompt`);
   595→
   596→  // Drain stdout/stderr to prevent backpressure
   597→  const stdoutPromise = new Response(proc.stdout as ReadableStream).text();
   598→  const stderrPromise = new Response(proc.stderr as ReadableStream).text();
   599→
   600→  let timeoutId: ReturnType<typeof setTimeout>;
   601→  const timeout = new Promise<never>((_, reject) => {
   602→    timeoutId = setTimeout(() => {
   603→      proc.kill();
   604→      reject(new Error("Metadata extraction timed out"));
   605→    }, METADATA_TIMEOUT_MS);
   606→  });
   607→
   608→  try {
   609→    await Promise.race([
   610→      proc.exited.then(async (code) => {
   611→        await stdoutPromise;
   612→        const stderr = await stderrPromise;
   613→        if (stderr.trim()) appendLog(agent, `[metadata stderr] ${stderr.trim()}`);
   614→        if (code !== 0) {
   615→          const hint = stderr.trim().split("\n").pop() || "";
   616→          throw new Error(`Metadata extraction exited with code ${code}${hint ? `: ${hint}` : ""}`);
   617→        }
   618→      }),
   619→      timeout,
   620→    ]);
   621→  } finally {
   622→    clearTimeout(timeoutId!);
   623→  }
   624→}

   633→/** Like teardownAgent but captures stderr lines into agent logs. */
   634→async function teardownAgentWithLogs(meta: SandboxMeta, cwd: string, agent: AgentState): Promise<TeardownResult> {
   635→  const proc = Bun.spawn([
   636→    "bash", join(SCRIPTS_DIR, "teardown-sandbox.sh"),
   637→    cwd, meta.worktreePath, meta.sandboxName, meta.tempBranch, meta.baseBranch, meta.model, meta.deerTmpDir,
   638→  ], { cwd, stdout: "pipe", stderr: "pipe" });
   639→
   640→  const stdoutPromise = new Response(proc.stdout as ReadableStream).text();
   641→  const stderrPromise = new Response(proc.stderr as ReadableStream).text();
   642→
   643→  const code = await proc.exited;
   644→  const stdout = await stdoutPromise;
   645→  const stderr = await stderrPromise;
   646→
   647→  // Surface the script's progress messages into agent logs
   648→  for (const line of stderr.trim().split("\n").filter(Boolean)) {
   649→    appendLog(agent, `[teardown] ${stripAnsi(line).trim()}`);
   650→  }
   651→
   652→  if (code !== 0) {
   653→    throw new Error(`Teardown failed (exit ${code}): ${stderr.trim().split("\n").pop()}`);
   654→  }
   655→
   656→  const lines = stdout.trim().split("\n").filter(Boolean);
   657→  const jsonLine = lines[lines.length - 1];
   658→  if (!jsonLine) throw new Error("Teardown script produced no output");
   659→  return JSON.parse(jsonLine) as TeardownResult;
   660→}

   662→function cleanupAgent(agent: AgentState, repoRoot: string) {
   663→  if (agent.proc) {
   664→    try { agent.proc.kill(); } catch { /* ignore */ }
   665→  }
   666→  if (agent.timer) clearInterval(agent.timer);
   667→  if (agent.meta) {
   668→    // Best-effort cleanup — fire-and-forget to avoid blocking the UI
   669→    const { sandboxName, worktreePath, tempBranch } = agent.meta;
   670→    (async () => {
   671→      try { await Bun.spawn(["docker", "sandbox", "rm", sandboxName]).exited; } catch { /* ignore */ }
   672→      try { await Bun.spawn(["git", "-C", repoRoot, "worktree", "remove", "--force", worktreePath]).exited; } catch { /* ignore */ }
   673→      try { await Bun.spawn(["git", "-C", repoRoot, "branch", "-D", tempBranch]).exited; } catch { /* ignore */ }
   674→    })();
   675→  }
   676→}

   678→/**
   679→ * Run metadata extraction + teardown for an agent.
   680→ * Handles its own errors — sets agent status to completed or failed.
   681→ */
   682→async function finalizeAgent(
   683→  agent: AgentState,
   684→  cwd: string,
   685→  setAgents: (updater: (prev: AgentState[]) => AgentState[]) => void,
   686→): Promise<void> {
   687→  if (!agent.meta) return;
   688→
   689→  // Kill the dead tmux session left over from the main Claude run
   690→  // (remain-on-exit keeps it alive). Cleaning up avoids stale lock files
   691→  // or session state that could interfere with the metadata invocation.
   692→  try {
   693→    await Bun.spawn([
   694→      "docker", "sandbox", "exec", agent.meta.sandboxName,
   695→      "tmux", "kill-session", "-t", "deer",
   696→    ], { stdout: "pipe", stderr: "pipe" }).exited;
   697→  } catch { /* ignore */ }
   698→
   699→  // Check if Claude wrote any code
   700→  const statusProc = Bun.spawn(
   701→    ["git", "-C", agent.meta.worktreePath, "status", "--porcelain"],
   702→    { stdout: "pipe", stderr: "pipe" },
   703→  );
   704→  const statusOut = await new Response(statusProc.stdout).text();
   705→  await statusProc.exited;
   706→  const changeCount = statusOut.trim().split("\n").filter(Boolean).length;
   707→  appendLog(agent, `[teardown] ${changeCount} file(s) changed in worktree`);
   708→
   709→  // Metadata extraction (non-fatal)
   710→  agent.lastActivity = "Generating PR metadata...";
   711→  setAgents((prev) => [...prev]);
   712→
   713→  try {
   714→    await startClaudeMetadata(agent.meta, agent);
   715→  } catch (err) {
   716→    appendLog(agent, `[warn] Metadata extraction failed: ${err instanceof Error ? err.message : String(err)}`);
   717→  }
   718→
   719→  // Teardown
   720→  agent.status = transition(agent.status, "TEARDOWN_START") ?? agent.status;
   721→  agent.lastActivity = "Committing and creating PR...";
   722→  setAgents((prev) => [...prev]);
   723→
   724→  try {
   725→    agent.result = await teardownAgentWithLogs(agent.meta, cwd, agent);
   726→    agent.status = transition(agent.status, "TEARDOWN_COMPLETE") ?? agent.status;
   727→
   728→    if (agent.result.prUrl) {
   729→      agent.lastActivity = "PR ready";
   730→    } else if (agent.transcript.length > 0) {
   731→      agent.transcriptPath = await persistTranscript(agent);
   732→      agent.lastActivity = "Answer ready — Enter to continue";
   733→    } else {
   734→      agent.lastActivity = "No changes";
   735→    }
   736→  } catch (err) {
   737→    agent.status = transition(agent.status, "ERROR") ?? "failed";
   738→    agent.error = err instanceof Error ? err.message : String(err);
   739→    agent.lastActivity = truncate(agent.error, 120);
   740→  } finally {
   741→    if (agent.timer) clearInterval(agent.timer);
   742→    agent.timer = null;
   743→    agent.proc = null;
   744→    await saveToHistory(agent, cwd);
   745→    setAgents((prev) => [...prev]);
   746→  }
   747→}

   749→/** Check the current state of a PR via `gh pr view`. */
   750→async function checkPrState(prUrl: string): Promise<"open" | "merged" | "closed" | null> {
   751→  try {
   752→    const proc = Bun.spawn(
   753→      ["gh", "pr", "view", prUrl, "--json", "state", "-q", ".state"],
   754→      { stdout: "pipe", stderr: "pipe" },
   755→    );
   756→    const code = await proc.exited;
   757→    if (code !== 0) return null;
   758→    const state = (await new Response(proc.stdout).text()).trim();
   759→    if (state === "MERGED") return "merged";
   760→    if (state === "CLOSED") return "closed";
   761→    return "open";
   762→  } catch {
   763→    return null;
   764→  }
   765→}

   769→/** @internal Exported for testing */
   770→export function parseNdjsonLine(line: string, agent: AgentState): boolean {
   771→  if (!line.trim()) return false;
   772→  let changed = false;
   773→
   774→  try {
   775→    const event = JSON.parse(line);
   776→
   777→    // Assistant text content
   778→    if (event.type === "assistant" && event.message?.content) {
   779→      for (const block of event.message.content) {
   780→        if (block.type === "text" && block.text) {
   781→          agent.lastActivity = truncate(block.text.replace(/\n/g, " ").trim(), 120);
   782→          appendLog(agent, block.text.trim());
   783→          agent.transcript.push(block.text);
   784→          changed = true;
   785→        }
   786→        if (block.type === "tool_use") {
   787→          const name = block.name || "tool";
   788→          const input = block.input ? JSON.stringify(block.input).slice(0, 100) : "";
   789→          agent.currentTool = `${name} ${input}`;
   790→          appendLog(agent, `[tool] ${name} ${input}`);
   791→          changed = true;
   792→        }
   793→      }
   794→    }
   795→
   796→    // Tool result
   797→    if (event.type === "result" && event.result) {
   798→      agent.lastActivity = truncate(String(event.result).replace(/\n/g, " ").trim(), 120);
   799→      changed = true;
   800→    }
   801→
   802→    // Content block delta (streaming)
   803→    if (event.type === "content_block_delta") {
   804→      if (event.delta?.type === "text_delta" && event.delta.text) {
   805→        agent.lastActivity = truncate(event.delta.text.replace(/\n/g, " ").trim(), 120);
   806→        changed = true;
   807→      }
   808→    }
   809→
   810→    // System message
   811→    if (event.type === "system" && event.message) {
   812→      appendLog(agent, `[system] ${event.message}`);
   813→    }
   814→  } catch {
   815→    // Not valid JSON — treat as plain text log
   816→    if (line.trim()) {
   817→      appendLog(agent, line.trim());
   818→      changed = true;
   819→    }
   820→  }
   821→
   822→  return changed;
   823→}

   825→/** Save a finished agent to the repo's history file. */
   826→async function saveToHistory(agent: AgentState, repoPath: string): Promise<void> {
   827→  if (agent.historical) return;
   828→  const status = agent.status as "completed" | "failed" | "cancelled";
   829→  const task: PersistedTask = {
   830→    taskId: agent.taskId,
   831→    prompt: agent.prompt,
   832→    status,
   833→    createdAt: new Date(Date.now() - agent.elapsed * 1000).toISOString(),
   834→    completedAt: new Date().toISOString(),
   835→    elapsed: agent.elapsed,
   836→    prUrl: agent.result?.prUrl ?? null,
   837→    finalBranch: agent.result?.finalBranch ?? null,
   838→    error: agent.error || null,
   839→    transcriptPath: agent.transcriptPath,
   840→    lastActivity: agent.lastActivity,
   841→  };
   842→  await upsertHistory(repoPath, task);
   843→}

   845→/** Convert a persisted task to a read-only AgentState for display. */
   846→function historicalAgent(task: PersistedTask, id: number): AgentState {
   847→  const wasInterrupted = task.status === "running";
   848→  return createAgentState({
   849→    id,
   850→    taskId: task.taskId,
   851→    prompt: task.prompt,
   852→    status: wasInterrupted ? "interrupted" : task.status,
   853→    elapsed: task.elapsed,
   854→    lastActivity: wasInterrupted ? "Interrupted — deer was closed" : task.lastActivity,
   855→    result: task.prUrl ? { finalBranch: task.finalBranch ?? "", prUrl: task.prUrl } : null,
   856→    error: task.error || "",
   857→    transcriptPath: task.transcriptPath,
   858→    historical: true,
   859→  });
   860→}
